fun somma(real a, real b): real
  real  result;
  ? 'Somma: ';
  result  := a + b;

  return result;
end fun;

fun moltiplica(real a, real b): real
  real  result;
  ? 'Moltiplicazione: ';
  if a = 0.0 or b = 0.0 then
    result := 0.0;
  else
    result := a * b;
  end if;
 return result;
end fun;

fun dividi(real a, real b): real
  real  result;
  ? 'Divisione: ';
  if b = 0.0 then
    result := 0.0;
  else
    result := a / b;
  end if;
  return result;
end fun;

fun sottrai(real a, real b): real
  real  result;
  ? 'Sottrazione: ';
  if a < b then
    result := 0.0;
  else
    result := a - b;
  end if;
  return result;
end fun;
